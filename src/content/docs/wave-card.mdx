<CodeBlock
  isLanguage={false} code={`
  @import "tw-animate-css";

@keyframes float {
  0%, 100% {
    transform: translateY(0);
  }
  50% {
    transform: translateY(-10px);
  }
}

@keyframes dataStream {
  0% {
    stroke-dashoffset: 20;
  }
  100% {
    stroke-dashoffset: 0;
  }
}

@keyframes schemaPulse {
  0%, 100% {
    transform: scale(1);
    opacity: 0.8;
  }
  50% {
    transform: scale(1.02);
    opacity: 1;
  }
}
`}
/>

### Props

| Prop          | Type     | Default    | Description                              |
| ------------- | -------- | ---------- | ---------------------------------------- |
| `tags`        | `string` | `Required` | Tag or label displayed above the title.  |
| `title`       | `string` | `Required` | Main title text displayed on the card.   |
| `description` | `string` | `Required` | Description text shown below the title.  |
| `buttonText`  | `string` | `Required` | Text displayed inside the action button. |
